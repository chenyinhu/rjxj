[root@jsetc lesson1]# cat 8-位运算.c
#include <stdio.h>

int main()
{
int num, p1, p2, i;

scanf("%d%d%d", &num, &p1, &p2);

if (p1 > p2 || p1 > 32 || p2 > 32)
{
printf("输入有误\n");
return -1;
}

for (i = 0; i < p2 - p1 + 1; i++)
{
printf("%d", (~((num >> (p2 - 1 - i)) & 1)) & 1);
}
printf("\n");

return 0;
}
