int less(int a, int b)
{
return (a > b) ? 1 : 0;
}

int greater(int a, int b)
{
return (a < b) ? 1 : 0;
}

void sort(int a[], int length, int (*p)(int, int)) //回调函数 把函数名作为函数的参数，可以修改函数的功能
{
int i, j;
for(i = 0; i < length - 1; i++)
{
for (j = 0; j < length - i - 1; j++)
{
//if (a[j] < a[j + 1])
if (p(a[j], a[j + 1]))
{
#if 0
int tmp = a[j];
a[j] = a[j + 1];
a[j + 1] = tmp;
#endif
a[j] = a[j] + a[j + 1];
a[j + 1] = a[j] - a[j + 1];
a[j] = a[j] - a[j + 1];
}
}
}

}

void print_array(int a[], int length)
{
int i;
for (i = 0; i < length; i++)
{
printf("%d ", a[i]);
}
printf("\n");

}

int main()
{
int i, j, a[5] = {0};

get_array(a, sizeof(a) / sizeof(a[0]));

sort(a, sizeof(a) / sizeof(a[0]), greater);

print_array(a, sizeof(a) / sizeof(a[0]));

return 0;
}

[root@jsetc lesson4]# cat 8-字符串排序.c
#include <stdio.h>
#include <stdlib.h>

void sort(char *s[], int size)
{
int i, j;

for (i = 0; i < size - 1; i++)
{
for (j = 0; j < size - i - 1; j++)
{
if (strcmp(s[j], s[j + 1]) > 0)
{
char *t = s[j];
s[j] = s[j + 1];
s[j + 1] = t;
}
}
}
}

int main(int argc, char *argv[])
{
int i;

sort(argv + 1, argc - 1);

for (i = 1; i < argc; i++)
{
printf("%s ", argv[i]);
}
printf("\n");

return 0;
}
