一维数组与二维数组初学

这里直接通过代码来解释：

 

[root@jsetc lesson2]# cat 1-一维数组.c
#include <stdio.h>

int g; //未初始化的全局变量是0

int main()
{
int k; //未初始化的局部变量是随机值
printf("%d\n", k);
printf("%d\n", g);

//int a[k] = {0}; //错误
//int a[5] = {1, 2, 3, 4, 5};
//int a[5];
//int a[5] = {1, 2};
//int a[] = {1, 2, 3, 4, 5, 6, 7};
int a[5] = {0};

for (k = 0; k < sizeof(a) / sizeof(a[0]); k++)//建议这样循环
{
printf("%d ", a[k]);
}
printf("\n");

printf("%p\n", &a[0]);
printf("%p\n", a); //数组名 数组首元素地址
printf("%p\n", &a); //&a是数组的地址

printf("%p\n", &a[0] + 1);
printf("%p\n", a + 1);
printf("%p\n", &a + 1);

return 0;
}
[root@jsetc lesson2]# cat 2-二维数组.c
#include <stdio.h>

int main()
{
//int a[3][4] = {{1, 2, 3, 4}, {2, 3, 4, 5}, {3, 4, 5, 6}};
//int a[3][4] = {{1, 2}, {2}, {3, 4, 5}};
//int a[][4] = {{1, 2, 3, 4}, {2, 3, 4, 5}, {3, 4, 5, 6}};
int a[3][4] = {{0}};

int i, j;
for (i = 0; i < 3; i++)
{
for (j = 0; j < 4; j++)
{
printf("%d ", a[i][j]);
}
printf("\n");
}

printf("%p\n", &a[0][0]);
printf("%p\n", a[0]); //首行首元素地址
printf("%p\n", a[1]);
printf("%p\n", *(a + 1)); //a[1] ===== *(a + 1) []等价于*()
printf("%p\n", a); //数组首行地址
printf("%p\n", &a); //数组的地址

printf("%p\n", &a[0][0] + 1);
printf("%p\n", a[0] + 1);
printf("%p\n", a[1] + 1);
printf("%p\n", a + 1);
printf("%p\n", &a + 1);

return 0;
}
