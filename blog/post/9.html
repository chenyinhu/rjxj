[root@jsetc lesson3]# cat 4-字符串翻转.c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/*
函数描述：字符串反转
函数参数：字符串的开始地址 字符串的结束地址
函数返回值：没有
*/
void reverse(char *begin, char *end)
{
int length = end - begin + 1;
int i;
for (i = 0; i < length / 2; i++)
{
char ch = *(begin + i);
*(begin + i)= *(end - i);
*(end - i) = ch;
}
}

int main()
{
char *str = (char *)malloc(sizeof(char) * 1024);
if (NULL == str)
{
printf("malloc filaure\n");
return -1;
}
char *s = str;

char ch;
int i = 0;
while ((ch = getchar()) != '\n')
{
*(str + i++) = ch;
}

reverse(str, str + strlen(str) - 1);

char *begin = str;
while (*str != '\0')
{
if (*str == ' ')
{
reverse(begin, str - 1);
begin = str + 1;
}
str++;
}
reverse(begin, str - 1);

printf("%s\n", s);

return 0;
}

[root@jsetc lesson3]# cat 8-字符串排序.c
#include <stdio.h>
#include <stdlib.h>

#define SIZE 5

void sort(char *s[])
{
int i, j;

for (i = 0; i < SIZE - 1; i++)
{
for (j = 0; j < SIZE - i - 1; j++)
{
if (strcmp(s[j], s[j + 1]) > 0)
{
char *t = s[j];
s[j] = s[j + 1];
s[j + 1] = t;
}
}
}
}

int main()
{
int i;
char *str[SIZE] = {0};

for (i = 0; i < SIZE; i++)
{
str[i] = (char *)malloc(sizeof(char) * 32);
scanf("%s", str[i]);
}

sort(str);

for (i = 0; i < SIZE; i++)
{
printf("%s ", str[i]);
}
printf("\n");

return 0;
}
