[root@jsetc lesson2]# cat 5-函数.c
#include <stdio.h>

void f1(void) //不能接受任意类型参数
{
}

void show() //没有返回值 没有形参（可以接受任意类型的参数）
{
printf("helloworld!\n");
}

int add(int x, int y) //形参 个数相同 类型相同 2、给形参分配空间 3、传参（值传递 地址传递）
{
int sum = x + y; //4、执行函数体
return sum; //5、返回 6、释放（栈）空间 （形参和局部变量）
}

int main()
{
//f1(1);
show();

int a = 1, b = 2;
int result = add(a, b); //实参 1、通过函数名找到函数的入口地址
printf("%d\n", result);
printf("%p\n", add); //函数名也是地址

return 0;
}

void sort(int a[], int length)
{
int i, j;
for(i = 0; i < length - 1; i++)
{
for (j = 0; j < length - i - 1; j++)
{
if (a[j] > a[j + 1])
{
#if 0
int tmp = a[j];
a[j] = a[j + 1];
a[j + 1] = tmp;
#endif
a[j] = a[j] + a[j + 1];
a[j + 1] = a[j] - a[j + 1];
a[j] = a[j] - a[j + 1];
}
}
}

}

void print_array(int a[], int length)
{
int i;
for (i = 0; i < length; i++)
{

}
